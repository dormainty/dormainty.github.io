zsh жҳҜд»Җд№Ҳпјҹ вҖӢ

zshпјҢжҲ– Z ShellпјҢжҳҜдёҖдёӘ Unix-Like зі»з»ҹпјҲеҰӮ macOS жҲ– LinuxпјүдёӢзҡ„ shell е‘Ҫд»ӨиЎҢи§ЈйҮҠеҷЁгҖӮ

е®ғж”ҜжҢҒејәеӨ§зҡ„иҮӘеҠЁиЎҘе…ЁиғҪеҠӣпјҢжӢҘжңүдё°еҜҢзҡ„жҸ’д»¶пјҢе…·жңүй«ҳеҸҜе®ҡеҲ¶жҖ§пјҢиҖҢдё”дёҺ bash е……еҲҶе…је®№гҖӮиҷҪз„¶пјҢе®ғдёҺ bash зӣёжҜ”пјҢиғҪеҠӣжӣҙеҠ ејәеӨ§пјҢдҪҶжҳҜе®ғеҚҙдҫқз„¶жҜ” bash жӣҙеҝ«гҖӮ

еҶҚиҖ…пјҢзӣёиҫғдәҺ bashпјҢzsh зҺ°еңЁзӨҫеҢәжӣҙеҠ жҙ»и·ғпјҢжҳҜдёҖдёӘиҝҳеңЁжҲҗй•ҝдёӯзҡ„йЎ№зӣ®гҖӮ

shell
# жҹҘзңӢеҪ“еүҚз»Ҳз«ҜshellжҳҜzshиҝҳжҳҜbash
echo $0
# еҲҮжҚўжҲҗbash
chsh -s /bin/bash
# еҲҮжҚўжҲҗzsh
chsh -s /bin/zsh

# зј–иҫ‘zshй…ҚзҪ®ж–Үд»¶
vim .zshrc
# дҝқеӯҳеҪ“еүҚй…ҚзҪ®
source ~/.zshrc
