js

<aa class="abc" v-model="test" ></aa> 

// aaеӯҗз»„д»¶е®һзҺ°дёҖпјҡ
<template>
  <div>
    <ul>
      <li>{{'йҮҢйқўзҡ„еҖјпјҡ'+ msg}}</li>

    </ul>
  </div>
</template>

<script>
  export default {
    model: {    // дҪҝз”Ёmodel
      prop: 'msg', //propеұһжҖ§е°ҶmsgдҪңдёәиҜҘз»„д»¶иў«дҪҝз”Ёж—¶пјҲжӯӨеӨ„дёәaaз»„д»¶иў«зҲ¶з»„д»¶и°ғз”Ёпјүv-modelиғҪеҸ–еҲ°зҡ„еҖјпјҢ
      event: 'cc' // eventдёӯзҡ„ccе°ұжҳҜзҲ¶з»„д»¶дёҠзҡ„иҮӘе®ҡд№үдәӢд»¶пјҢз”ЁжқҘжӣҙж–°зҲ¶з»„д»¶дёҠзҡ„testеҖј
    },
    props: {
      msg: ''
    },
    methods: {
      fn2 () {
        this.$emit('cc', this.msg+2)
      }
    }
  }
</script>

// aaеӯҗз»„д»¶е®һзҺ°ж–№жі•дәҢпјҡ
