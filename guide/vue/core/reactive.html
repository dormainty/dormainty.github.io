TIP

дҪңз”Ёпјҡе®ҡд№үдёҖдёӘе“Қеә”ејҸеҜ№иұЎ

иҜӯжі•пјҡlet е“Қеә”ејҸеҜ№иұЎ = reactive(жәҗеҜ№иұЎ)

иҝ”еӣһеҖјпјҡдёҖдёӘ Proxy зҡ„е®һдҫӢеҜ№иұЎпјҢз®Җз§°пјҡе“Қеә”ејҸеҜ№иұЎ

reactive е®ҡд№үзҡ„е“Қеә”ејҸж•°жҚ®жҳҜ ж·ұеұӮж¬Ў зҡ„

reactive еҲӣе»әеҜ№иұЎзұ»еһӢзҡ„е“Қеә”ејҸж•°жҚ® вҖӢ

<template>
  <div>{{ obj.age }}</div>
  <button @click="increment">еўһеҠ </button>
</template>

<script lang="ts" setup>
import { reactive } from 'vue'
let obj = reactive({
  age: 18
})
const increment = () => {
  obj.age++
}
</script>

  get() {
    console.log('get')

  },

    console.log('set', val)
    data.checklist = data.list.map((item) => val)
  }
})
</script>
